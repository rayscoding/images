# JavaScript Code for Slider Functionality

```js
let currentSlide = 0;
const slides = document.querySelectorAll('.slide');
const indicators = document.querySelectorAll('.indicator');
const thumbnails = document.querySelectorAll('.thumbnail');
const progressBar = document.getElementById('progressBar');
const playPauseIcon = document.getElementById('playPauseIcon');

let autoplay = true;
const slideIntervalTime = 5000; // 5 seconds
let slideInterval;

function showSlide(index) {
    if (index >= slides.length) index = 0;
    if (index < 0) index = slides.length - 1;

    // Remove active classes
    slides.forEach(slide => slide.classList.remove('active', 'prev'));
    indicators.forEach(ind => ind.classList.remove('active'));
    thumbnails.forEach(thumbnail => thumbnail.classList.remove('active'));

    // Set classes for current and previous slide for animation
    slides[currentSlide].classList.add('prev');
    slides[index].classList.add('active');
    indicators[index].classList.add('active');
    thumbnails[index].classList.add('active');

    currentSlide = index;
    resetProgressBar();
}

function changeSlide(direction) {
    showSlide(currentSlide + direction);
}

function goToSlide(index) {
    showSlide(index);
}

function resetProgressBar() {
    progressBar.style.width = '0%';
}

function updateProgressBar() {
    let width = 0;
    const interval = setInterval(() => {
        if (width >= 100) {
            clearInterval(interval);
            if (autoplay) changeSlide(1);
        } else {
            width += 1;
            progressBar.style.width = width + '%';
        }
    }, slideIntervalTime / 100);
}

function startAutoplay() {
    autoplay = true;
    playPauseIcon.classList.remove('fa-play');
    playPauseIcon.classList.add('fa-pause');
    updateProgressBar();
}

function stopAutoplay() {
    autoplay = false;
    playPauseIcon.classList.remove('fa-pause');
    playPauseIcon.classList.add('fa-play');
    progressBar.style.width = '0%';
}

function toggleAutoplay() {
    if (autoplay) {
        stopAutoplay();
    } else {
        startAutoplay();
    }
}

// Initialize slider
showSlide(currentSlide);
startAutoplay();

// Restart progress bar on slide change
document.querySelector('.main-slider').addEventListener('transitionend', () => {
    if (autoplay) {
        resetProgressBar();
        updateProgressBar();
    }
});
````

### JavaScript Explanation

* `showSlide(index)`: Ye function slides ko show karta hai aur unka active class manage karta hai.
* `changeSlide(direction)`: Aage ya peeche slide badalne ke liye.
* `goToSlide(index)`: Direct kisi bhi slide pe jump karne ke liye thumbnails ya indicators se.
* Progress bar ko update karne ke liye `updateProgressBar` function use hota hai jo 5 seconds me fill hota hai.
* `startAutoplay` aur `stopAutoplay` play/pause control dete hain.
* Slider load hone ke baad pehla slide show hota hai aur autoplay shuru ho jata hai.
